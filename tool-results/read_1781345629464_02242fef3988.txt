   117→function OwnershipCalc() {
   118→  const [year, setYear] = useState(2022)
   119→  const [engineCc, setEngineCc] = useState(1400)
   120→  const [fuelType, setFuelType] = useState('Benzin')
   121→  const [avgFuel, setAvgFuel] = useState(7)
   122→  const [yearlyKm, setYearlyKm] = useState(15000)
   123→  const [price, setPrice] = useState(1000000)
   124→
   125→  const currentYear = new Date().getFullYear()
   126→  const age = currentYear - year
   127→
   128→  // MTV hesaplama
   129→  const engineBand = engineCc <= 1300 ? '0-1300' : engineCc <= 1600 ? '1301-1600' : engineCc <= 1800 ? '1601-1800' : engineCc <= 2000 ? '1801-2000' : engineCc <= 2500 ? '2001-2500' : '2501+'
   130→  const ageBand = age <= 3 ? '1' : age <= 5 ? '2' : '3'
   131→  const mtvTable: Record<string, number> = {
   132→    '0-1300_1': 1313, '1301-1600_1': 2280, '1601-1800_1': 4560, '1801-2000_1': 7297, '2001-2500_1': 14595, '2501+_1': 25542,
   133→    '0-1300_2': 985, '1301-1600_2': 1710, '1601-1800_2': 3420, '1801-2000_2': 5473, '2001-2500_2': 10946, '2501+_2': 19156,
   134→    '0-1300_3': 656, '1301-1600_3': 1140, '1601-1800_3': 2280, '1801-2000_3': 3648, '2001-2500_3': 7297, '2501+_3': 12771,
   135→  }
   136→  const mtv = mtvTable[`${engineBand}_${ageBand}`] || 1500
   137→  const kasko = Math.round(price * (age <= 2 ? 0.055 : age <= 5 ? 0.04 : 0.03))
   138→  const fuelPrices: Record<string, number> = { 'Benzin': 42.5, 'Dizel': 40.0, 'LPG': 22.0, 'Hibrit': 42.5, 'Elektrik': 0 }
   139→  const yearlyFuel = fuelType === 'Elektrik' ? Math.round(15 * 2.5 * yearlyKm / 100) : Math.round(avgFuel * (fuelPrices[fuelType] || 42.5) * yearlyKm / 100)
   140→  const yearlyMaintenance = age <= 3 ? 5000 : age <= 7 ? 8000 : 12000
   141→  const yearlyTotal = mtv + kasko + yearlyFuel + yearlyMaintenance
   142→  const monthlyTotal = Math.round(yearlyTotal / 12)
   143→
   144→  return (
   145→    <div>
   146→      <div className="grid grid-cols-2 sm:grid-cols-3 gap-3 mb-4">
   147→        <div>
   148→          <label className="text-[10px] text-slate-500 block mb-1">Model Yılı</label>
   149→          <input type="number" value={year} onChange={e => setYear(Number(e.target.value))} min={2000} max={2026} className="w-full bg-white border border-slate-200 rounded-lg px-2.5 py-1.5 text-xs text-slate-900 focus:outline-none focus:border-[#e37224]" />
   150→        </div>
   151→        <div>
   152→          <label className="text-[10px] text-slate-500 block mb-1">Motor Hacmi (cc)</label>
   153→          <input type="number" value={engineCc} onChange={e => setEngineCc(Number(e.target.value))} min={800} max={5000} step={100} className="w-full bg-white border border-slate-200 rounded-lg px-2.5 py-1.5 text-xs text-slate-900 focus:outline-none focus:border-[#e37224]" />
   154→        </div>
   155→        <div>
   156→          <label className="text-[10px] text-slate-500 block mb-1">Yakıt Tipi</label>
   157→          <select value={fuelType} onChange={e => setFuelType(e.target.value)} className="w-full bg-white border border-slate-200 rounded-lg px-2.5 py-1.5 text-xs text-slate-900 focus:outline-none focus:border-[#e37224] appearance-none">
   158→            {['Benzin', 'Dizel', 'LPG', 'Hibrit', 'Elektrik'].map(f => <option key={f} value={f}>{f}</option>)}
   159→          </select>
   160→        </div>
   161→        <div>
   162→          <label className="text-[10px] text-slate-500 block mb-1">Ort. Yakıt (lt/100km)</label>

   164→        </div>
   165→        <div>
   166→          <label className="text-[10px] text-slate-500 block mb-1">Yıllık KM</label>
   167→          <input type="number" value={yearlyKm} onChange={e => setYearlyKm(Number(e.target.value))} min={1000} max={100000} step={1000} className="w-full bg-white border border-slate-200 rounded-lg px-2.5 py-1.5 text-xs text-slate-900 focus:outline-none focus:border-[#e37224]" />
   168→        </div>
   169→        <div>
   170→          <label className="text-[10px] text-slate-500 block mb-1">Araç Fiyatı (TL)</label>
   171→          <input type="number" value={price} onChange={e => setPrice(Number(e.target.value))} min={100000} max={50000000} step={50000} className="w-full bg-white border border-slate-200 rounded-lg px-2.5 py-1.5 text-xs text-slate-900 focus:outline-none focus:border-[#e37224]" />

   174→      <div className="grid grid-cols-2 sm:grid-cols-4 gap-3">
   175→        {[

   178→          { label: 'Yakıt', amount: yearlyFuel },

   183→            <span className="text-sm font-bold text-slate-800">{item.amount.toLocaleString('tr-TR')} TL</span>
   184→          </div>
   185→        ))}
   186→      </div>
   187→      <div className="mt-4 bg-gradient-to-r from-[#e37224]/10 to-[#c85e1a]/10 border border-[#f09040]/30 rounded-xl p-4 text-center">
   188→        <span className="text-xs text-slate-500 block">Yıllık Toplam Maliyet</span>

   316→function ChipSelect({ options, selected, onToggle, maxShow }: { options: string[]; selected: string[]; onToggle: (v: string) => void; maxShow?: number }) {
   317→  const [showAll, setShowAll] = useState(false)
   318→  const displayed = maxShow && !showAll ? options.slice(0, maxShow) : options
   319→  return (
   320→    <div>
   321→      <div className="flex flex-wrap gap-1.5">
   322→        {displayed.map(o => (
   323→          <button key={o} onClick={() => onToggle(o)}
   324→            className={`text-xs px-2.5 py-1 rounded-full border transition-all ${selected.includes(o) ? 'bg-[#fff8f0] border-[#e37224] text-[#e37224]' : 'bg-slate-100 border-slate-200 text-slate-500 hover:border-slate-300'}`}>
   325→            {o}
   326→          </button>
   327→        ))}
   328→        {maxShow && options.length > maxShow && !showAll && (
   329→          <button onClick={() => setShowAll(true)} className="text-[10px] px-2 py-1 text-[#e37224] hover:text-[#c85e1a]">+{options.length - maxShow} daha</button>
   330→        )}
   331→        {maxShow && showAll && (
   332→          <button onClick={() => setShowAll(false)} className="text-[10px] px-2 py-1 text-[#e37224] hover:text-[#c85e1a]">Daralt</button>
   333→        )}

   335→    </div>
